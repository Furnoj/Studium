\section{Altersbestimmung} \label{sec:Altersbestimung}

\begin{itemize}
\item $N(t) = N_0 \cdot e^{- \lambda \cdot t}$ 
\item $\lambda = \frac{ln(2)}{t_{\frac{1}{2}}}$
\item $t = \frac{t_{\frac{1}{2}}}{ln(2)} \cdot ln (\frac{N_t}{N_0})$
\end{itemize}
